What: In D3.js, Transitions allow you to animate changes to DOM elements smoothly over time.

Why: Transitions make visualizations dynamic, engaging, and help users understand data changes.

Where: Use transitions for bars, circles, paths, colors, opacity, transforms, and any visual property changes.

Basic D3.js Transitions Examples

1. Simple Bar Height Animation


// Basic bar height animation
function simpleBarAnimation() {
    const data: number[] = [10, 20, 30, 15, 25];
    
    const container = d3.select('#simple-bar-transition');
    container.selectAll('*').remove();

    // Create bars
    container.selectAll('div.bar')
        .data(data)
        .enter()
        .append('div')
        .style('display', 'inline-block')
        .style('width', '30px')
        .style('margin', '2px')
        .style('background-color', 'steelblue')
        .style('height', '0px')
        .style('vertical-align', 'bottom')
        // Add transition
        .transition()
        .duration(1000)
        .style('height', d => d * 10 + 'px');
}

2. Color Change Animation


// Simple color transition
function colorChangeAnimation() {
    const svg = d3.select('#color-transition');
    svg.selectAll('*').remove();

    const circle = svg.append('circle')
        .attr('cx', 100)
        .attr('cy', 75)
        .attr('r', 40)
        .attr('fill', 'blue');

    // Change color smoothly
    circle.transition()
        .duration(2000)
        .attr('fill', 'red');
}

3. Position Movement Animation

5. Fade In/Out Animation


// Fade animation
function fadeAnimation() {
    const container = d3.select('#fade-transition');
    container.selectAll('*').remove();

    const box = container.append('div')
        .style('width', '100px')
        .style('height', '100px')
        .style('background-color', 'orange')
        .style('margin', '20px auto')
        .style('opacity', 0);

    // Fade in
    box.transition()
        .duration(1000)
        .style('opacity', 1);
}
