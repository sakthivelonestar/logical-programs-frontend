Post<T>(apiEndPoint: string, data?: unknown): Observable<T> {
  return this.httpClient.post<T>(`${environment.url}${apiEndPoint}`, data, { withCredentials: true });
}

PostById<T>(apiEndPoint: string, id: number, data: unknown): Observable<T> {

}

Get<T>(apiEndPoint: string, queryParams?: unknown): Observable<T> {
  let qp = this.utilityService.jsonToQueryString(queryParams);
  return this.httpClient.get<T>(`${environment.url}${apiEndPoint}${qp}`);
}

GetByHeaders<T>(apiEndPoint: string, queryParams?: unknown): Observable<T> {
  let qp = this.utilityService.jsonToQueryString(queryParams);
  return this.httpClient.get<T>(`${environment.url}${apiEndPoint}${qp}`, { responseType: 'blob' as 'json' });
}

}

        import { ApiService } from '../../services/api.service';
        import { ApiList } from '../../core/varibles/api-list';
        import {  HttpErrorResponse } from '@angular/common/http';
         constructor(private apiService: ApiService) { }

Post(): void {
  const data = { totalFPN: 1000, actualForThePeriod: 500, percentageFPN: 50, availableBudget: 2000 };
  this.apiService.Post<IRes>(ApiList.getFinancialCardData, data).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}

PostById(id: number): void {
  const data = { totalFPN: 1500, actualForThePeriod: 700, percentageFPN: 60, availableBudget: 2500 };
  this.apiService.PostById<IRes>(ApiList.getFinancialCardData, id, data).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}

Get(queryParams?: unknown): void {
  const params = { filter: 'active', limit: 10 };
  this.apiService.Get<IRes>(ApiList.getFinancialCardData, params).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}
