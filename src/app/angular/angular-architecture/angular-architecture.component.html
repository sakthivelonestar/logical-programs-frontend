ng new my-app --standalone --style=scss --routing

cd my-app
ng serve --open

Step 2 — API Service and Endpoints

📌 api.service.ts — CRUD operations


import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from '../../environments/environment';

@Injectable({
  providedIn: 'root'
})
constructor(private httpClient: HttpClient) { }

Post<any>(apiEndPoint: string, data?: unknown): Observable<any> {
  return this.httpClient.post<any>(`${environment.url}${apiEndPoint}`, data, { withCredentials: true });
}

PostById<any>(apiEndPoint: string, id: number, data: unknown): Observable<any> {
  return this.httpClient.post<any>(`${environment.url}${apiEndPoint}/${id}`, data);
}

Get<T>(apiEndPoint: string, queryParams?: unknown): Observable<T> {
  let qp = this.utilityService.jsonToQueryString(queryParams);
  return this.httpClient.get<T>(`${environment.url}${apiEndPoint}${qp}`);
}

Put<T>(apiEndPoint: string, data?: unknown): Observable<T> {
  return this.httpClient.put<T>(`${environment.url}${apiEndPoint}`, data);
}

Delete<T>(apiEndPoint: string, id: number): Observable<T> {
  return this.httpClient.delete<T>(`${environment.url}${apiEndPoint}/${id}`);
}

📌utilityService.ts — Function used in API service to convert json to query string for headers


 import { Injectable } from '@angular/core';  
  @Injectable({
      providedIn:  'root'
  })
 export class UtilityService {
   jsonToQueryString(json: any) {
    if(json) {
      return '?' +  Object.keys(json).map(function(key) {
         return  encodeURIComponent(key) + '=' +encodeURIComponent(json[key]);
      }).join('&'); 
     } else {
     return '' 
    }
  }
}

import { ApiService } from '../../services/api.service';
import { ApiList } from '../../core/varibles/api-list';
import { HttpErrorResponse } from '@angular/common/http';

constructor(private apiService: ApiService) { }

Post(): void {
  const data = { totalFPN: 1000, actualForThePeriod: 500, percentageFPN: 50, availableBudget: 2000 };
  this.apiService.Post<IRes>(ApiList.getFinancialCardData, data).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}

15 Important Interview Questions on Angular Architecture & Project Setup

  1. What are the prerequisites for creating an Angular project?

    You need Node.js (v16 or higher), npm or yarn package manager, and Angular CLI installed globally.

    Basics
  2. How do you create a new Angular project?

    Run npm install -g @angular/cli to install Angular CLI, then create a project using ng new my-app, configure routing and styles, and start the server with ng serve -o.

    Setup
  3. What are the key configuration files in an Angular project?

    angular.json (build and project config), package.json (dependencies), tsconfig.json (TypeScript settings), .browserslistrc (browser compatibility), and src/environments/ (environment configs).

    Configuration
  4. What is the role of main.ts in Angular?

    main.ts is the entry point of the application. It bootstraps the root module (AppModule) using platformBrowserDynamic().bootstrapModule().

    Core
  5. What are the core building blocks of Angular architecture?

    Modules, Components, Services, Directives, Pipes, and Routing are the key building blocks of Angular applications.

    Architecture
  6. What is the purpose of Angular modules?

    Modules group related functionality. The root module is AppModule, and feature modules are created to organize large-scale applications.

    Modules
  7. How should you structure a large-scale Angular application?

    Use core/ (singleton services, guards, interceptors), shared/ (reusable components), features/ (feature modules), layout/ (UI structure), assets/ (static files), and environments/ (config).

    Best Practices
  8. What is the role of Angular CLI in project development?

    Angular CLI simplifies project setup, scaffolding, building, serving, and testing through commands like ng new, ng serve, and ng build.

    CLI
  9. How do you generate Angular components, services, and modules?

    Use CLI commands like ng generate component [name], ng generate service [name], and ng generate module [name].

    Development
  10. What is the role of Dependency Injection in Angular?

    Dependency Injection (DI) allows services and resources to be provided where needed, ensuring reusability and maintainability.

    Core
  11. How is routing handled in Angular?

    Angular uses RouterModule to configure routes. It supports lazy loading, route guards, and child routes.

    Routing
  12. How do you handle environment-specific settings in Angular?

    Use src/environments/environment.ts for development and environment.prod.ts for production, configured in angular.json.

    Configuration
  13. What are Angular services, and why are they important?

    Services contain business logic and data access. They are injected into components and other services using Angular’s DI system.

    Services
  14. How do you test Angular applications?

    Use ng test for unit tests (Jasmine & Karma) and ng e2e for end-to-end tests (Protractor or Cypress).

    Testing
  15. What are some best practices for Angular project setup?

    Use feature modules, lazy loading, shared modules for reusability, environment-specific configs, consistent naming conventions, and follow Angular style guide.

    Best Practices
