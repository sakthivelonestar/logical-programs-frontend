📦 Project Setup


 # Install Angular CLI
npm install -g @angular/cli

# Create new Angular 18 app
ng new my-app --standalone --style=scss --routing

# Start server
cd my-app
ng serve --open

# Generate component & service
ng generate component financial
ng generate service services/api

🏗️ Angular 18 Architecture

  • 🔧 Standalone Components: No NgModule needed, easy to test
  • 📡 Signals: Reactive state, fine-grained updates, type-safe
  • ⚙️ App Config: Centralized config, functional bootstrap, easy DI
  • 🚄 SSR & Hydration: Server-side rendering, SEO friendly, fast loading

Post<T>(apiEndPoint: string, data?: unknown): Observable<T> {
  return this.httpClient.post<T>(`${environment.url}${apiEndPoint}`, data, { withCredentials: true });
}

PostById<T>(apiEndPoint: string, id: number, data: unknown): Observable<T> {
  return this.httpClient.post<T>(`${environment.url}${apiEndPoint}/${id}`, data);
}

Get<T>(apiEndPoint: string, queryParams?: unknown): Observable<T> {
  let qp = this.utilityService.jsonToQueryString(queryParams);
  return this.httpClient.get<T>(`${environment.url}${apiEndPoint}${qp}`);
}

GetByHeaders<T>(apiEndPoint: string, queryParams?: unknown): Observable<T> {
  let qp = this.utilityService.jsonToQueryString(queryParams);
  return this.httpClient.get<T>(`${environment.url}${apiEndPoint}${qp}`, { responseType: 'blob' as 'json' });
}

PostByHeaders<T>(apiEndPoint: string, data: any, queryParams?: unknown): Observable<T> {
  let qp = this.utilityService.jsonToQueryString(queryParams);
  return this.httpClient.post<T>(`${environment.url}${apiEndPoint}${qp}`, data, { responseType: 'blob' as 'json' });
}

GetById<T>(apiEndPoint: string, id: number | string, queryParams?: unknown): Observable<T> {
  let qp = this.utilityService.jsonToQueryString(queryParams);
  return this.httpClient.get<T>(`${environment.url}${apiEndPoint}/${id}${qp}`);
}

Put<T>(apiEndPoint: string, data?: unknown): Observable<T> {
  return this.httpClient.put<T>(`${environment.url}${apiEndPoint}`, data);
}

PutById<T>(apiEndPoint: string, id: number, data: unknown): Observable<T> {
  return this.httpClient.put<T>(`${environment.url}${apiEndPoint}/${id}`, data);
}

Delete<T>(apiEndPoint: string, id: number): Observable<T> {
  return this.httpClient.delete<T>(`${environment.url}${apiEndPoint}/${id}`);
}

API Service

The ApiService provides CRUD operations (Create, Read, Update, Delete) for RESTful API communication, using TypeScript generics. The ApiList enum defines the endpoint for financial card data.

📌 financial.component.ts — API call examples 


        import { ApiService } from '../../services/api.service';
        import { ApiList } from '../../core/varibles/api-list';
        import {  HttpErrorResponse } from '@angular/common/http';
         constructor(private apiService: ApiService) { }

Post(): void {
  const data = { totalFPN: 1000, actualForThePeriod: 500, percentageFPN: 50, availableBudget: 2000 };
  this.apiService.Post<IRes>(ApiList.getFinancialCardData, data).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}

PostById(id: number): void {
  const data = { totalFPN: 1500, actualForThePeriod: 700, percentageFPN: 60, availableBudget: 2500 };
  this.apiService.PostById<IRes>(ApiList.getFinancialCardData, id, data).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}

Get(queryParams?: unknown): void {
  const params = { filter: 'active', limit: 10 };
  this.apiService.Get<IRes>(ApiList.getFinancialCardData, params).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}

GetByHeaders(queryParams?: unknown): void {
  const params = { format: 'blob' };
  this.apiService.GetByHeaders<IRes>(ApiList.getFinancialCardData, params).subscribe({
    next: (res) => { this.financialCardData = res.status ? res.data : []; },
    error: (err: HttpErrorResponse) => { this.financialCardData = []; }
  });
}
